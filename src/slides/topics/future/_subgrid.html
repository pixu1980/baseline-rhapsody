Subgrid ✅ Widely Available

/* Parent grid */
.grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  gap: 1rem;
}

/* Child inherits parent's grid */
.card {
  display: grid;
  grid-template-rows: subgrid;
  grid-row: span 3;
}

/* All cards align perfectly */
.card h3 { grid-row: 1; }
.card img { grid-row: 2; }
.card p { grid-row: 3; }
