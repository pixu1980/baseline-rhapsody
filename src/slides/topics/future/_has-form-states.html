:has() + Form States ✅ Widely Available

/* Style form based on validity */
form:has(:invalid) {
  border-color: red;
}

form:has(:invalid)::before {
  content: "⚠️ Please fix errors";
  color: red;
}

/* Style parent based on checked state */
.option:has(input:checked) {
  background: #3b82f6;
  color: white;
}

/* Disabled state styling */
.field:has(input:disabled) {
  opacity: 0.5;
  cursor: not-allowed;
}
