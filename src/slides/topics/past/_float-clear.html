The Infamous Clearfix Hack

/* The classic clearfix */
.clearfix::before,
.clearfix::after {
  content: "";
  display: table;
}

.clearfix::after {
  clear: both;
}

/* IE6/7 fallback */
.clearfix {
  *zoom: 1;
}
