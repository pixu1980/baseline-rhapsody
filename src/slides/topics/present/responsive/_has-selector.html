:has() Selector ✅ Widely Available

/* Parent selector! */
.card:has(img) {
  display: grid;
  grid-template-columns: 200px 1fr;
}

/* Style parent based on children */
form:has(:invalid) {
  border-color: red;
}

/* Conditional styling */
.container:has(> .alert) {
  padding-top: 3rem;
}

/* Style siblings */
li:has(+ li) {
  border-bottom: 1px solid;
}
