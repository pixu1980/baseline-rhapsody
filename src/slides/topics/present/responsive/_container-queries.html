Container Queries ✅ Widely Available

/* Define container */
.container {
  container-type: inline-size;
  container-name: card;
}

/* Query the container, not viewport */
@container card (min-width: 400px) {
  .card-title {
    font-size: 2rem;
  }
  
  .card-layout {
    display: grid;
    grid-template-columns: 1fr 2fr;
  }
}

/* Component truly responsive! */
