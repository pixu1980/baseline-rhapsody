@scope ✅ Widely Available

/* Scope styles to subtree */
@scope (.card) {
  h2 { 
    font-size: 1.5rem; 
    color: blue;
  }
  
  p { 
    line-height: 1.6; 
  }
}

/* Scope with lower boundary */
@scope (.article) to (.sidebar) {
  /* Styles apply in .article 
     but NOT in .sidebar */
  a { color: inherit; }
}
