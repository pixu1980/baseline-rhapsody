@layer reset {
  * { margin: 0; padding: 0; }
}

@layer base {
  h1 { font-size: 2rem; }
}

@layer components {
  .button { 
    padding: 1rem; 
    background: blue; 
  }
}

/* Utilities always win */
@layer utilities {
  .mt-0 { margin-top: 0 !important; }
}

1. reset
2. base
3. components
4. utilities
