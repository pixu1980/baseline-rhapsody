    Basic Usage

    /* Load variable font */
@font-face {
  font-family: 'Inter Variable';
  src: url('Inter-Variable.woff2') format('woff2-variations');
  font-weight: 100 900;
  font-stretch: 75% 125%;
}

/* Use standard properties */
.text {
  font-family: 'Inter Variable';
  font-weight: 450; /* Any value 100-900 */
  font-stretch: 110%;
}

/* Fine-grained control */
.custom {
  font-variation-settings: 
    'wght' 550,
    'wdth' 105;
}

    Custom Axes

    /* Fonts can define custom axes */
@font-face {
  font-family: 'Recursive';
  src: url('Recursive-Variable.woff2') format('woff2-variations');
}

.custom-axis {
  font-family: 'Recursive';
  
  /* Custom axes (font-specific) */
  font-variation-settings:
    'MONO' 0,    /* 0 = sans, 1 = mono */
    'CASL' 0,    /* 0 = linear, 1 = casual */
    'slnt' 0,    /* -15 to 0 degrees */
    'CRSV' 0.5;  /* 0 to 1 cursive */
}

/* Predefined styles */
.casual {
  font-variation-settings: 'CASL' 1;
}

.monospace {
  font-variation-settings: 'MONO' 1;
}

    Animating Variable Fonts

    /* Animate weight */
@keyframes weight-wave {
  0%, 100% { font-weight: 300; }
  50% { font-weight: 700; }
}

.animated-text {
  font-family: 'Inter Variable';
  animation: weight-wave 2s ease-in-out infinite;
}

/* Animate width on hover */
.expand {
  font-family: 'Inter Variable';
  font-stretch: 100%;
  transition: font-stretch 0.3s;
}

.expand:hover {
  font-stretch: 125%;
}

/* Complex animation */
@keyframes morph {
  0% {
    font-variation-settings: 'wght' 300, 'wdth' 75;
  }
  50% {
    font-variation-settings: 'wght' 700, 'wdth' 125;
  }
  100% {
    font-variation-settings: 'wght' 300, 'wdth' 75;
  }
}

    Responsive Typography

    /* Adjust weight based on viewport */
.responsive-heading {
  font-family: 'Inter Variable';
  
  /* Thinner on mobile */
  font-weight: 400;
}

@media (min-width: 768px) {
  .responsive-heading {
    /* Bolder on desktop */
    font-weight: 600;
  }
}

/* Fluid weight with clamp */
.fluid-weight {
  font-weight: clamp(300, 20vw, 800);
}

/* Optical sizing for better readability */
.optical {
  font-family: 'Source Serif Variable';
  font-optical-sizing: auto;
  /* Automatically adjusts for font-size */
}

    Feature Detection

    /* Check for variable font support */
@supports (font-variation-settings: normal) {
  .text {
    font-family: 'Inter Variable';
    font-variation-settings: 'wght' 450;
  }
}

/* Fallback for older browsers */
@supports not (font-variation-settings: normal) {
  .text {
    font-family: 'Inter', sans-serif;
    font-weight: 400;
  }
}
    // JavaScript detection
if ('fontVariationSettings' in document.body.style) {
  // Variable fonts supported
  document.body.classList.add('variable-fonts');
}
