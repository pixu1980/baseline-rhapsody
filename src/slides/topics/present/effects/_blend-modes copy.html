    Mix Blend Mode Examples

    /* Blend element with background */
.blend-text {
  mix-blend-mode: multiply;
  color: white;
  /* Text darkens the background */
}

/* Duotone effect */
.duotone {
  position: relative;
  background: #667eea;
}

.duotone img {
  mix-blend-mode: multiply;
  opacity: 0.8;
}

/* Knockout text effect */
.knockout {
  mix-blend-mode: difference;
  color: white;
  /* Inverts colors underneath */
}

    Background Blend Mode

    /* Blend multiple backgrounds */
.hero {
  background-image: 
    linear-gradient(rgba(102, 126, 234, 0.8), rgba(118, 75, 162, 0.8)),
    url('background.jpg');
  background-blend-mode: multiply;
  background-size: cover;
}

/* Color overlay effect */
.color-overlay {
  background: 
    linear-gradient(#ff6b6b, #ff6b6b),
    url('image.jpg');
  background-blend-mode: color;
  background-size: cover;
}

/* Texture effect */
.textured {
  background:
    url('noise.png'),
    linear-gradient(135deg, #667eea, #764ba2);
  background-blend-mode: overlay;
}

    Practical Example: Image Effects

    /* Duotone image effect */
.duotone-image {
  position: relative;
  background: linear-gradient(45deg, #667eea, #f093fb);
}

.duotone-image img {
  display: block;
  width: 100%;
  mix-blend-mode: multiply;
}

/* Add highlights */
.duotone-image::after {
  content: '';
  position: absolute;
  inset: 0;
  background: linear-gradient(45deg, #667eea, #f093fb);
  mix-blend-mode: screen;
  opacity: 0.3;
}

/* Color pop effect */
.color-pop {
  filter: grayscale(100%);
  mix-blend-mode: luminosity;
}

.color-pop:hover {
  filter: grayscale(0%);
  mix-blend-mode: normal;
  transition: all 0.5s ease;
}

    Text Effects with Blend Modes

    /* Knockout text (inverted) */
.knockout-text {
  font-size: 4rem;
  font-weight: bold;
  color: white;
  mix-blend-mode: difference;
  /* Text inverts background colors */
}

/* Gradient text blend */
.gradient-blend {
  background: linear-gradient(90deg, #667eea, #f093fb);
  -webkit-background-clip: text;
  background-clip: text;
  color: transparent;
  mix-blend-mode: multiply;
}

/* Glowing text effect */
.glow-text {
  color: #667eea;
  text-shadow: 0 0 20px #667eea;
  mix-blend-mode: screen;
}

    Creative Composition

    /* Layered blend effects */
.complex-blend {
  position: relative;
}

.complex-blend::before {
  content: '';
  position: absolute;
  inset: 0;
  background: radial-gradient(circle, #667eea, transparent);
  mix-blend-mode: color-dodge;
}

.complex-blend::after {
  content: '';
  position: absolute;
  inset: 0;
  background: linear-gradient(45deg, transparent, #764ba2);
  mix-blend-mode: overlay;
}

/* Neon effect */
.neon {
  background: #000;
  color: #0ff;
  text-shadow: 
    0 0 10px #0ff,
    0 0 20px #0ff,
    0 0 30px #0ff;
  mix-blend-mode: screen;
}

    Performance Considerations

    • ✅ Blend modes are GPU-accelerated
    • ⚠️ Can create new stacking contexts
    • ✅ Use will-change for animations
    • ⚠️ Test on mobile devices
    /* Optimize animated blends */
.animated-blend {
  will-change: mix-blend-mode;
  mix-blend-mode: normal;
  transition: mix-blend-mode 0.3s;
}

.animated-blend:hover {
  mix-blend-mode: multiply;
}
