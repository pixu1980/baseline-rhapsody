/* Mask with image or gradient */
.element {
  mask-image: linear-gradient(
    to bottom,
    black 0%,
    transparent 100%
  );
}

/* SVG masks */
.complex-mask {
  mask-image: url('mask.svg');
  mask-size: cover;
  mask-repeat: no-repeat;
}

/* Combine with clip-path */
.clipped {
  clip-path: polygon(
    0% 0%, 100% 0%, 100% 75%, 0% 100%
  );
}
