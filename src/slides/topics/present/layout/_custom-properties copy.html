    Fallback Values

    /* Provide fallback if variable is not defined */
.element {
  color: var(--text-color, #000000);
  font-size: var(--font-size, 16px);
}

/* Chain multiple fallbacks */
.element {
  color: var(--brand-color, var(--primary-color, #667eea));
}

/* Use with calc() */
.element {
  margin: calc(var(--spacing, 1rem) * 2);
}

    Scoped Variables

    /* Global variables */
:root {
  --color-primary: blue;
}

/* Component-scoped variables */
.card {
  --card-padding: 1rem;
  --card-bg: white;
  
  padding: var(--card-padding);
  background: var(--card-bg);
}

.card.large {
  --card-padding: 2rem; /* Override in larger variant */
}

/* Context-specific overrides */
.sidebar .card {
  --card-bg: #f5f5f5;
}

    Dynamic Updates with JavaScript

    // Get variable value
const root = document.documentElement;
const primaryColor = getComputedStyle(root)
  .getPropertyValue('--primary-color');

// Set variable value
root.style.setProperty('--primary-color', '#ff6b6b');

// Update per element
element.style.setProperty('--item-width', '300px');

// Example: Color picker
colorPicker.addEventListener('change', (e) => {
  root.style.setProperty('--accent-color', e.target.value);
});

    Advanced: Spacing System

    :root {
  /* Base unit */
  --space-unit: 0.5rem;
  
  /* Spacing scale */
  --space-xs: calc(var(--space-unit) * 1);   /* 0.5rem */
  --space-sm: calc(var(--space-unit) * 2);   /* 1rem */
  --space-md: calc(var(--space-unit) * 4);   /* 2rem */
  --space-lg: calc(var(--space-unit) * 8);   /* 4rem */
  --space-xl: calc(var(--space-unit) * 16);  /* 8rem */
}

.component {
  padding: var(--space-md);
  margin-bottom: var(--space-lg);
  gap: var(--space-sm);
}
