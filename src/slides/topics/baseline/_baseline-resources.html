Feature Detection in CSS

/* @supports - Check if feature is available */
@supports (container-type: inline-size) {
  /* Container queries supported */
  .card {
    container-type: inline-size;
  }
  
  @container (min-width: 400px) {
    .card-content {
      display: flex;
    }
  }
}

/* Fallback for browsers without support */
@supports not (container-type: inline-size) {
  .card {
    /* Use media queries instead */
  }
  
  @media (min-width: 600px) {
    .card-content {
      display: flex;
    }
  }
}

Advanced @supports Queries

/* AND condition */
@supports (display: grid) and (gap: 1rem) {
  .layout {
    display: grid;
    gap: 1rem;
  }
}

/* OR condition */
@supports (display: flex) or (display: -webkit-flex) {
  .container {
    display: flex;
  }
}

/* NOT condition */
@supports not (display: grid) {
  .layout {
    /* Flexbox fallback */
    display: flex;
    flex-wrap: wrap;
  }
}

/* Complex conditions */
@supports (backdrop-filter: blur(10px)) and 
          (not (background-clip: text)) {
  .glass {
    backdrop-filter: blur(10px);
  }
}

Feature Detection in JavaScript

// Check CSS property support
if (CSS.supports('container-type', 'inline-size')) {
  console.log('Container queries supported!');
}

// Check with full declaration
if (CSS.supports('backdrop-filter: blur(10px)')) {
  document.body.classList.add('glassmorphism-support');
}

// Check multiple conditions
const hasModernCSS = 
  CSS.supports('container-type', 'inline-size') &&
  CSS.supports('selector(:has(*))') &&
  CSS.supports('(color: oklch(0.5 0.2 180))');

if (hasModernCSS) {
  console.log('All modern CSS features available!');
}

// Check for API support
if ('popover' in HTMLElement.prototype) {
  console.log('Popover API supported!');
}

if (document.startViewTransition) {
  console.log('View Transitions API supported!');
}

Progressive Enhancement Strategy

Layer Your CSS Features

  1. Base layer - Use Widely Available features only
  2. Enhancement layer - Add Newly Available features with @supports
  3. Experimental layer - Add Limited features for modern browsers
/* 1. Base (Widely Available) */
.card {
  display: flex;
  flex-direction: column;
  padding: 1rem;
  background: white;
  border-radius: 8px;
}

/* 2. Enhancement (Newly Available) */
@supports (container-type: inline-size) {
  .card {
    container-type: inline-size;
  }
  
  @container (min-width: 400px) {
    .card {
      flex-direction: row;
    }
  }
}

/* 3. Experimental (Limited) */
@supports (anchor-name: --my-anchor) {
  .tooltip {
    position: absolute;
    position-anchor: --my-anchor;
    bottom: anchor(top);
  }
}

Analytics & Monitoring

// Track feature support in your analytics
const features = {
  containerQueries: CSS.supports('container-type', 'inline-size'),
  hasSelector: CSS.supports('selector(:has(*))'),
  popoverAPI: 'popover' in HTMLElement.prototype,
  viewTransitions: !!document.startViewTransition,
  anchorPositioning: CSS.supports('anchor-name', '--test'),
};

// Send to your analytics
analytics.track('CSS Feature Support', features);

// Or use with error tracking
if (!features.containerQueries) {
  console.warn('Container queries not supported, using fallback');
  // Optionally report to error tracking service
}
